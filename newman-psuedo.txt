async function RunNewman(warmup: true | false): return empty Promise{
    return new Promise(resolve, reject){
        run Newman{
            collection: from argument
            iteration count: if warmup 100 | from argument
        }.on request(error | null, any args){
            push to results{
                iteration++,
                responseTime from measured responseTime
            }
        }.on done(error | null, any args){
            if warmup{
                return reject | resolve
            }

            pause for 500ms to let program finish reading memory

            get memory reads from memory reader script

            if memory reads != iterations{
                reject promise and exit process since there are not enough memory reads
            }

            map results into new array object and add memory usage to it

            if all results are correct{
                create report with name as output
            } else{
                reject promise and exit program
            }
        }
    }
}
